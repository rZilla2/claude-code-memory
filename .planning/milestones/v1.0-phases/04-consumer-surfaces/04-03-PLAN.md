    3. Define output shape:
       ```typescript
       interface ContextResult {
         target: { id: string; sourcePath: string; headingPath: string; text: string; indexedAt: string };
         prev: { id: string; sourcePath: string; headingPath: string; text: string; indexedAt: string } | null;
         next: { id: string; sourcePath: string; headingPath: string; text: string; indexedAt: string } | null;
       }
       ```

    4. Export registration function:
       ```typescript
       export function registerGetContextTool(server: McpServer, table: lancedb.Table): void
       ```

    5. Tool registration:
       ```typescript
       server.tool(
         'get_context',
         'Retrieve a chunk by ID with its neighboring chunks from the same source file for surrounding context',
         { chunkId: z.string().describe('The chunk ID from a search_memory result') },
         async ({ chunkId }) => { ... }
       );
       ```

    6. Handler implementation:
       a. Query LanceDB for the target chunk by ID:
          ```typescript
          const SELECT_COLS = ['id', 'text', 'source_path', 'heading_path', 'indexed_at'];
          const targetRows = await table.query()
            .where(`id = '${chunkId.replace(/'/g, "''")}'`)
            .select(SELECT_COLS)
            .limit(1)
            .toArray();

          if (targetRows.length === 0) {
            return {
              content: [{ type: 'text', text: JSON.stringify({ error: `Chunk not found: ${chunkId}` }) }],
              isError: true,
            };
          }
          const target = targetRows[0] as unknown as ChunkRow;
          ```

       b. Query all chunks from the same source file:
          ```typescript
          const siblingRows = await table.query()
            .where(`source_path = '${target.source_path.replace(/'/g, "''")}'`)
            .select(SELECT_COLS)
            .toArray();
          ```

       c. Sort by heading_path to establish positional order:
          ```typescript
          const siblings = (siblingRows as unknown as ChunkRow[])
            .sort((a, b) => a.heading_path.localeCompare(b.heading_path));
          ```

       d. Find target index and extract neighbors:
          ```typescript
          const targetIdx = siblings.findIndex(s => s.id === chunkId);
          const prevChunk = targetIdx > 0 ? siblings[targetIdx - 1] : null;
          const nextChunk = targetIdx < siblings.length - 1 ? siblings[targetIdx + 1] : null;
          ```

       e. Format and return:
          ```typescript
          const format = (row: ChunkRow) => ({
            id: row.id,
            sourcePath: row.source_path,
            headingPath: row.heading_path,
            text: row.text,
            indexedAt: new Date(Number(row.indexed_at)).toISOString(),
          });

          const result: ContextResult = {
            target: format(target),
            prev: prevChunk ? format(prevChunk) : null,
            next: nextChunk ? format(nextChunk) : null,
          };

          return { content: [{ type: 'text', text: JSON.stringify(result) }] };
          ```

    CRITICAL: No console.log anywhere. Only logger for error reporting.
    CRITICAL: SQL-escape single quotes in chunkId and source_path to prevent injection.
