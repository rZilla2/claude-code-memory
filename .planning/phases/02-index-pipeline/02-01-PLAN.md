```typescript
import { glob } from 'fast-glob';
import type { Config } from '../types.js';

export async function scanVault(config: Config): Promise<string[]> {
  // Build extension pattern: '**/*.md' for default, or union for multiple
  const extPattern = config.includeExtensions.length === 1
    ? `**/*${config.includeExtensions[0]}`
    : `**/*{${config.includeExtensions.join(',')}}`;

  const ignore = [
    '**/.obsidian/**',
    '**/node_modules/**',
    '**/*.icloud',
    ...config.ignorePaths.map(p => `**/${p}/**`),
  ];

  return glob(extPattern, {
    cwd: config.vaultPath,
    absolute: true,
    ignore,
    followSymbolicLinks: false,
  });
}
```
